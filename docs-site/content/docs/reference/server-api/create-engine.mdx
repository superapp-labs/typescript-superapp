`createEngine` accepts a single configuration object and returns an engine instance. This page documents every available option.

```typescript
import { createEngine } from '@superapp/backend'

const engine = createEngine({
  connections: {
    main: 'postgres://localhost:5432/mydb',
  },
})
```

## `connections`

Each connection maps a name to a database URL or config object. The engine infers the database type from the URL protocol — no explicit `type` field needed.

```typescript
connections: {
  // Postgres — detected from postgres:// protocol
  main: 'postgres://localhost:5432/mydb',

  // MySQL — detected from mysql:// protocol
  warehouse: 'mysql://localhost:3306/warehouse',

  // SQLite — detected from .db file extension
  legacy: './legacy.db',

  // CSV — detected from { directory } shape
  imports: { directory: './data/imports/' },
}
```

### Type Inference

| URL / Config | Inferred Type |
|---|---|
| `postgres://...` or `postgresql://...` | PostgreSQL |
| `mysql://...` | MySQL |
| File path (`.db`, `.sqlite`) or `file:...` | SQLite |
| `libsql://...` or Turso URL | SQLite (Turso) |
| `{ directory: string }` | CSV |

### Examples

```typescript
// Single Postgres
connections: {
  main: process.env.PG_URL!,
}

// Multiple databases
connections: {
  main: process.env.PG_URL!,
  warehouse: process.env.MYSQL_URL!,
  legacy: './legacy.db',
  imports: { directory: './data/imports/' },
}
```

## `permissions`

Permission definitions using Drizzle-like syntax. Each permission targets a table, declares which roles have access, and configures operations (`select`, `insert`, `update`, `delete`) individually.

```typescript
permissions: {
  view_own_orders: {
    table: 'main.orders',
    roles: ['viewer', 'editor', 'admin'],
    select: {
      columns: ['id', 'amount', 'status', 'created_at'],
      where: { customer_id: { $eq: '$user.customer_id' } },
    },
  },
}
```

See [Permission Object](/docs/reference/server-api/permission-object) for the complete shape reference.

## `actions`

Typed server-side functions with inline roles. Each action defines its allowed roles directly.

```typescript
actions: {
  incrementStock: {
    roles: ['warehouse_manager', 'admin'],
    input: z.object({ productId: z.string(), amount: z.number().positive() }),
    output: z.object({ id: z.string(), stock: z.number() }),
    run: async ({ user, db }, { productId, amount }) => {
      const [updated] = await db
        .update(products)
        .set({ stock: sql`stock + ${amount}` })
        .where(eq(products.id, productId))
        .returning({ id: products.id, stock: products.stock })
      return updated
    },
  },
},
```

See [Actions](/docs/backend/actions) for full documentation.

## Full Example

```typescript
import { createEngine } from '@superapp/backend'
import { betterAuthProvider } from '@superapp/backend/auth/better-auth'

const engine = createEngine({

  // ────────────────────────────────────────
  // Core
  // ────────────────────────────────────────

  // 'programmatic' = define config in code
  // 'admin_ui'     = manage config from the admin dashboard
  mode: 'programmatic',

  // Where superapp stores its own tables (sessions, audit logs, admin config)
  // SQLite file path or a Turso URL
  superapp_db: process.env.TURSO_URL ?? './superapp.db',

  // Admin API key — required for admin_ui mode and /admin endpoints
  // Always use an env var, never hardcode
  masterKey: process.env.SUPERAPP_MASTER_KEY!,

  // ────────────────────────────────────────
  // Database Connections
  // ────────────────────────────────────────

  // Each key becomes a namespace prefix (e.g. main.orders, warehouse.events)
  // Type is auto-detected from the URL protocol
  connections: {
    main: process.env.PG_URL!,
    warehouse: process.env.MYSQL_URL!,
  },

  // ────────────────────────────────────────
  // Authentication
  // ────────────────────────────────────────

  // Auth provider — if omitted, the engine runs in open mode (no auth)
  auth: betterAuthProvider({
    secret: process.env.AUTH_SECRET!,
    userTable: {
      table: 'main.users',
      matchOn: { column: 'id', jwtField: 'id' },
      columns: ['id', 'email', 'name'],
    },
  }),

  // ────────────────────────────────────────
  // JWT Verification
  // ────────────────────────────────────────

  jwt: {
    // Allowed signing algorithms (common: 'HS256', 'RS256', 'ES256')
    algorithms: ['RS256'],

    // Expected 'iss' claim — tokens from other issuers are rejected
    issuer: 'https://auth.myapp.com',

    // Expected 'aud' claim — tokens for other audiences are rejected
    audience: 'https://api.myapp.com',

    // Seconds of clock drift allowed when checking token expiry
    clockSkewSeconds: 5,
  },

  // ────────────────────────────────────────
  // Limits & Rate Limiting
  // ────────────────────────────────────────

  limits: {
    // Cap on rows returned per query (client limit is capped to this)
    maxRows: 10_000,

    // How deep relations can nest: orders → items → product = depth 3
    maxRelationDepth: 3,

    // How deep $and/$or/$not can nest inside each other
    maxFilterNesting: 5,

    // Total number of conditions across all nesting levels
    maxFilterConditions: 50,

    // Max JSON body size (accepts '512kb', '1mb', '2mb', etc.)
    maxRequestBodySize: '1mb',

    // Kill queries that run longer than this (in ms)
    queryTimeout: 30_000,

    // Requests per minute per authenticated user
    rateLimitPerUser: 200,

    // Requests per minute per IP (applies to unauthenticated too)
    rateLimitPerIP: 500,
  },

  // ────────────────────────────────────────
  // Audit Logging
  // ────────────────────────────────────────

  audit: {
    // Turn audit logging on or off
    enabled: true,

    // Log the SQL statement that was executed
    logQuery: true,

    // Log query parameters (useful for debugging, may contain sensitive data)
    logParams: false,

    // Log how long each query took (in ms)
    logDuration: true,

    // Log which user ran each query
    logUser: true,

    // Log when a user is denied access (helps detect unauthorized attempts)
    logDenied: true,

    // Log admin UI actions (permission changes, connection edits)
    logAdminActions: true,

    // Auto-delete logs older than this (accepts '30d', '90d', '1y', etc.)
    retention: '90d',

    // Strip emails and phone numbers from log entries
    piiRedaction: true,
  },

  // ────────────────────────────────────────
  // Network Security
  // ────────────────────────────────────────

  security: {
    cors: {
      // Which origins can make requests (never use '*' in production)
      origin: ['https://myapp.com'],

      // Send Access-Control-Allow-Credentials: true
      credentials: true,
    },

    // Only these IPs can access /admin (all others are blocked)
    adminIpAllowlist: ['10.0.0.0/8'],
  },

  // ────────────────────────────────────────
  // Schema Introspection
  // ────────────────────────────────────────

  schemaEndpoint: {
    // Expose GET /schema for introspection
    enabled: true,

    // Require this token in the Authorization header (optional)
    token: process.env.SCHEMA_API_TOKEN,
  },

  // ────────────────────────────────────────
  // Permissions
  // ────────────────────────────────────────

  // Each key is a snake_case permission slug
  // Roles are declared inline on each permission
  permissions: {
    view_own_orders: {
      table: 'main.orders',
      roles: ['viewer', 'warehouse_manager'],
      select: {
        columns: ['id', 'amount', 'status', 'created_at'],
        where: { customer_id: { $eq: '$user.customer_id' } },
      },
    },
  },

  // ────────────────────────────────────────
  // Actions
  // ────────────────────────────────────────

  // Typed server-side functions callable via POST /actions/{name}
  actions: {
    incrementStock: {
      roles: ['warehouse_manager'],
      input: z.object({ productId: z.string(), amount: z.number().positive() }),
      output: z.object({ id: z.string(), stock: z.number() }),
      run: async ({ user, db }, { productId, amount }) => {
        const [updated] = await db
          .update(products)
          .set({ stock: sql`stock + ${amount}` })
          .where(eq(products.id, productId))
          .returning({ id: products.id, stock: products.stock })
        return updated
      },
    },
  },
})
```
